#include<vector>
#include<queue>
#include<iostream>
#include<map>

using namespace std;
struct Pos{ 
    int y;
    int x;

    bool operator<(const Pos& other) const {
        if (y == other.y) {
            return x < other.x;
        }
        return y < other.y;
    }

};

int solution(vector<vector<int> > maps)
{
    int answer = 0;
    queue<Pos> q;
    
    Pos startPoint = { 0 , 0};
    Pos endPoint = { (int)maps.size() - 1, (int)maps[0].size() - 1 };
    
    Pos dir[4] = {
        {-1, 0}, {1, 0}, {0, -1}, {0, 1}
    };
    bool found = false;
    
    q.push(startPoint);
    // 💡 시작점 maps[0][0]은 이미 1이므로 따로 세팅할 필요 없음!
    
    while(!q.empty()){
        Pos qCurrent = q.front();
        q.pop();
        
        if(qCurrent.x == endPoint.x && qCurrent.y == endPoint.y){
            found = true;
            break;
        }
        
        for(int i=0; i < 4; ++i){
            int nextX = qCurrent.x + dir[i].x;
            int nextY = qCurrent.y + dir[i].y;
            
            if(nextX < 0 || nextX >= maps[0].size() || nextY < 0 || nextY >= maps.size()){
                continue;
            }
            
            // 💡 maps 값이 1인 곳만 방문 (벽 0과 이미 방문한 곳인 2 이상을 한 번에 걸러냄)
            if (maps[nextY][nextX] == 1) {
                maps[nextY][nextX] = maps[qCurrent.y][qCurrent.x] + 1;
                q.push({nextY, nextX});
            }
        }
    }
    
    if (found) {
        answer = maps[endPoint.y][endPoint.x];
    }
    
    return found ? answer : -1;
}
